// 收藏数据 let favorites = JSON.parse(localStorage.getItem("fav") || "[]"); // 搜索 function getKey() { return document.getElementById("search").value.toLowerCase(); } // ⭐ 收藏 function toggleFav(name, url) { const index = favorites.findIndex(f => f.url === url); if (index > -1) favorites.splice(index, 1); else favorites.push({ name, url }); localStorage.setItem("fav", JSON.stringify(favorites)); render(); } // ⭐ 是否收藏 function isFav(url) { return favorites.some(f => f.url === url); } // 🔥 高亮 function highlight(text, key) { if (!key) return text; return text.replaceAll(key, `${key}`); } // 🔥 递归树 function createNode(node) { const div = document.createElement("div"); const title = document.createElement("div"); title.className = "node-title"; title.innerText = "📁 " + (node.category || node.subcategory || "节点"); const content = document.createElement("div"); content.className = "node-content"; title.onclick = () => { if (content.style.maxHeight) { content.style.maxHeight = null; } else { content.style.maxHeight = content.scrollHeight + "px"; } }; const key = getKey(); // items if (node.items) { node.items.forEach(i => { if (key && !i.name.toLowerCase().includes(key) ) return; const row = document.createElement("div"); const star = document.createElement("span"); star.className = "star"; star.innerText = isFav(i.url) ? "⭐" : "☆"; star.onclick = () => toggleFav(i.name, i.url); const a = document.createElement("a"); a.href = i.url; a.target = "_blank"; a.innerHTML = highlight(i.name, key); row.appendChild(star); row.appendChild(a); content.appendChild(row); }); } // children（递归） if (node.children) { node.children.forEach(child => { content.appendChild(createNode(child)); }); } div.appendChild(title); div.appendChild(content); return div; } // 左侧导航 function renderSidebar() { const sb = document.getElementById("sidebar"); sb.innerHTML = "📂 导航"; data.forEach(d => { const item = document.createElement("div"); item.innerText = d.category; item.onclick = () => { document.getElementById("app").innerHTML = ""; document.getElementById("app").appendChild(createNode(d)); }; sb.appendChild(item); }); } // 主渲染 function render() { const app = document.getElementById("app"); app.innerHTML = ""; data.forEach(d => { app.appendChild(createNode(d)); }); } renderSidebar(); render();
